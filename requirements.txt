# --- Data Handling ---
# pandas and numpy are the backbone of everything.
# pyarrow is needed for reading/writing parquet files
# which is what we use for the Silver layer.
pandas==2.2.2
numpy==1.26.4
pyarrow==16.0.0

# --- Machine Learning ---
# scikit-learn handles preprocessing and baseline models.
# xgboost is our main model — works really well on
# tabular claims data, better than deep learning here.
# imbalanced-learn deals with the fact that fraud cases
# are rare — maybe 1-2% of all claims.
scikit-learn==1.4.2
xgboost==2.0.3
imbalanced-learn==0.12.2

# --- Experiment Tracking ---
# MLflow tracks every training run so we always know
# which model is in production and why it was chosen.
mlflow==2.12.1

# --- Pipeline Orchestration ---
# Airflow runs the whole pipeline on a schedule.
# Fair warning — Airflow has a lot of dependencies.
# If install is slow that's normal, just let it run.
apache-airflow==2.9.1

# --- API Serving ---
# FastAPI for the prediction endpoint.
# Uvicorn is the server that actually runs it.
# Pydantic handles request validation automatically.
fastapi==0.111.0
uvicorn==0.29.0
pydantic==2.7.1

# --- Model Packaging ---
# BentoML packages the model for deployment.
# Makes SageMaker deployment much cleaner.
bentoml==1.2.10

# --- AWS ---
# boto3 is the standard AWS SDK for Python.
# sagemaker is the higher level wrapper on top of it.
boto3==1.34.84
sagemaker==2.215.0

# --- Monitoring ---
# Evidently generates drift reports on the fly.
# We run it daily via Airflow to catch
# when billing patterns start shifting.
evidently==0.4.30

# --- Data Validation ---
# Great Expectations checks data quality before
# anything gets into the training pipeline.
# Saved us from bad data more than once.
great-expectations==0.18.15

# --- Testing ---
# pytest for unit tests, pytest-cov for coverage reports.
# Aim for at least 70% coverage before calling it done.
pytest==8.2.0
pytest-cov==5.0.0

# --- Code Quality ---
# ruff is faster than flake8 and handles formatting too.
# mypy catches type errors before they blow up in prod.
ruff==0.4.4
mypy==1.10.0

# --- Misc Utilities ---
# python-dotenv loads API keys from .env file.
# Never hardcode credentials — always use .env.
# pyyaml reads config files, requests makes HTTP calls.
python-dotenv==1.0.1
pyyaml==6.0.1
requests==2.31.0
